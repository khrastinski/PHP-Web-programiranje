<?php
// Trenutni dan i sat
$dan = date("N");   // 1 (pon) – 7 (ned)
$sat = date("G");   // 0 – 23

$status = "Zatvoreno";

// Popis državnih praznika (format: m-d)
$praznici = [
    "01-01", // Nova godina
    "01-06", // Sveta tri kralja
    "05-01", // Praznik rada
    "06-22", // Dan antifašističke borbe
    "08-05", // Dan pobjede
    "08-15", // Velika Gospa
    "11-01", // Svi sveti
    "12-25"  // Božić
];

// Provjera praznika
$danas = date("m-d");
if (in_array($danas, $praznici)) {
    $status = "Zatvoreno (praznik)";
}
else {
    // Ponedjeljak – petak
    if ($dan >= 1 && $dan <= 5) {
        if ($sat >= 8 && $sat < 20) {
            $status = "Otvoreno";
        }
    }
    // Subota
    else if ($dan == 6) {
        if ($sat >= 9 && $sat < 14) {
            $status = "Otvoreno";
        }
    }
    // Nedjelja
    else if ($dan == 7) {
        $status = "Zatvoreno (nedjelja)";
    }
}
?>

    <meta charset="UTF-8">
    <title>Vježba 9 – Radno vrijeme</title>

        }
        .closed {
            color: red;
            font-weight: bold;
        }
    </style>
</head>
<body>

<div class="box">
    <p>Trenutno vrijeme: <strong><?= date("d.m.Y. H:i") ?></strong></p>

    <p>Status:
        <?php if ($status == "Otvoreno"): ?>
            <span class="open"><?= $status ?></span>
        <?php else: ?>
            <span class="closed"><?= $status ?></span>
        <?php endif; ?>
    </p>

    <hr>

        Ponedjeljak – Petak: 8 – 20<br>

    </p>
</div>
